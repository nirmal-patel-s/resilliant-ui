async function retryFetch(fn, retries = 3) {
  for (let i = 0; i < retries; i++) {
    try {
      return await fn();
    } catch (e) {
      await new Promise(res => setTimeout(res, 500 * (i + 1)));
    }
  }
  throw new Error('Failed after retries');
}
